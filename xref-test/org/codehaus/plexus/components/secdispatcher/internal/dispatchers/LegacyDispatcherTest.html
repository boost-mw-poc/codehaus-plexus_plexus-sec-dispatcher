14  package org.codehaus.plexus.components.secdispatcher.internal.dispatchers;
15  
16  import java.util.Map;
17  
18  import org.junit.jupiter.params.ParameterizedTest;
19  import org.junit.jupiter.params.provider.ValueSource;
20  
21  import static org.junit.jupiter.api.Assertions.assertEquals;
22  
23  public class LegacyDispatcherTest {
24      /**
25       * Test values created with Maven 3.9.9.
26       * <p>
27       * master password: "masterpassword"
28       * password: "password"
29       */
30      @ParameterizedTest
31      @ValueSource(
32              strings = {
33                  "src/test/legacy/legacy-settings-security-1.xml",
34                  "src/test/legacy/legacy-settings-security-2.xml"
35              })
36      void smoke(String xml) {
37          System.setProperty("settings.security", xml);
38          LegacyDispatcher legacyDispatcher = new LegacyDispatcher();
39          // SecDispatcher "un decorates" the PW
40          String cleartext = legacyDispatcher.decrypt("L6L/HbmrY+cH+sNkphnq3fguYepTpM04WlIXb8nB1pk=", Map.of(), Map.of());
41          assertEquals("password", cleartext);
42      }
43  }
